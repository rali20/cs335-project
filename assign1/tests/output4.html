package main

import (
  fmt
  strings
)

func main() {
  // Create a tic-tac-toe board.
  game := [][]string{
    []string{95, 95, 95},
    []string{95, 95, 95},
    []string{95, 95, 95},
  }

  // The players take turns.
  game[0][0] = 88
  game[2][2] = 79
  game[2][0] = 88
  game[1][0] = 79
  game[0][2] = 88

  printBoard(game)
}

func printBoard(s [][]string) {
  for i := 0; i < len(s); i++ {
    fmt.Printf(%s\n, strings.Join(s[i], 32))
  }
}
